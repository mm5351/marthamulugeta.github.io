Plotly

library(tidyverse)
library(viridis)
library(p8105.datasets)
library(plotly)
library(patchwork)
set.seed(1)
data("rest_inspec")
inspection_df = 
  rest_inspec %>% 
  select(score, boro, grade, violation_code) %>% 
  filter(!is.na(score), 
         boro != "Missing", 
         grade %in% c("A", "B", "C")) %>% 
  rename(Violation = violation_code,
         Grade = grade,
         Score = score,
         Borough = boro) %>% 
  mutate(Borough = str_to_title(Borough)) %>% 
  sample_n(5000)
inspection_df %>% 
  select(Borough, Score) %>%
  plot_ly(y = ~Score, color = ~Borough, type = "box",
          colors = "Set2")
inspection_df %>% 
  select(Grade, Borough) %>% 
  count(Grade, Borough) %>% 
  plot_ly(x = ~Grade, y = ~n, color = ~Borough, type = "bar", colors = "Set2")
inspection_df %>% 
  select(Score, Grade) %>% 
  count(Score, Grade) %>% 
  plot_ly(x = ~n, y = ~Score, color = ~Grade, type = "scatter", colors = "Set2")
## No scatter mode specifed:
##   Setting the mode to markers
##   Read more about this attribute -> https://plot.ly/r/reference/#scatter-mode
inspection_df %>%
  select(Violation) %>% 
  count(Violation) %>% 
  mutate(Violation = fct_reorder(Violation, n)) %>% 
  plot_ly(x = ~Violation, y = ~n, color = ~Violation, type = "bar", colors = "Set2")
